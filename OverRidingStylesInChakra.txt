Chakra styling / themes

Change styles by style props

import {extendtheme} from '@chakra/theme-tools'


export const myNewTheme = extendTheme({
  colors: {
  	primary: blue
  	secondary: green
  	highlight: yellow
  },
  components: {
    Input: {
      sizes: {
        md: {
          borderRadius: none
        }
      }
    }
  }
    
})

<ChakraProvider resetCSS theme={myNewTheme}>

</ChakraProvider>
